```javascript
async exportPositions() {
  const positions = await api.getPositions();
  const json = JSON.stringify(positions, null, 2);
  const blob = new Blob([json], { type: 'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `zedsu-positions-${Date.now()}.json`;
  a.click();
  URL.revokeObjectURL(url);
  window.ShellApi?.Toast.success('Positions exported');
},

async importPositions(fileInput) {
  const file = fileInput.files[0];
  if (!file) return;
  const text = await file.text();
  let positions;
  try {
    positions = JSON.parse(text);
  } catch {
    window.ShellApi?.Toast.error('Invalid JSON file');
    return;
  }
  let imported = 0;
  for (const pos of positions) {
    if (pos.name && (pos.x !== undefined || pos.y !== undefined)) {
      await api.setPosition(pos.name, pos);
      imported++;
    }
  }
  window.ShellApi?.Toast.success(`Imported ${imported} positions`);
  load(container);
},

async testAll() {
  const positions = await api.getPositions();
  const enabled = positions.filter(p => p.enabled !== false);
  if (enabled.length === 0) {
    window.ShellApi?.Toast.warning('No enabled positions to test');
    return;
  }

  // Create progress area
  const progressArea = document.createElement('div');
  progressArea.className = 'position-test-all-progress';
  progressArea.innerHTML = `<div class="progress-text">Testing ${enabled.length} positions...</div><div class="progress-detail" id="test-all-detail">Starting...</div>`;

  // Find the Quick Test section and add progress
  const quickTestSection = container.querySelector('.position-test-all-progress')
    || container.querySelector('[data-test-progress]');
  if (quickTestSection) {
    quickTestSection.replaceWith(progressArea);
  }

  for (let i = 0; i < enabled.length; i++) {
    const pos = enabled[i];
    document.getElementById('test-all-detail').textContent =
      `[${i+1}/${enabled.length}] Testing: ${pos.name}`;
    const result = await api.resolvePosition(pos.name);
    const status = result.success ? '✓' : '✗';
    document.getElementById('test-all-detail').textContent =
      `[${i+1}/${enabled.length}] ${status} ${pos.name} — ${result.success ? `(${result.x?.toFixed(0)}, ${result.y?.toFixed(0)})` : result.error}`;
    await new Promise(r => setTimeout(r, 300));
  }

  progressArea.innerHTML = `<div class="progress-text">✓ All ${enabled.length} positions tested</div>`;
  setTimeout(() => progressArea.remove(), 3000);
  window.ShellApi?.Toast.success('All positions tested');
},
```
