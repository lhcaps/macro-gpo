use std::sync::Mutex as StdMutex;
use tauri::tray::{TrayIcon, TrayIconBuilder, TrayIconEvent, MouseButton, MouseButtonState};
use tauri::menu::{Menu, MenuItem, PredefinedMenuItem};
use tauri::image::Image;

// ============================================================================
// Tray Manager
// ============================================================================

/// Tray icon state
#[derive(Debug, Clone, Copy, PartialEq)]
pub enum TrayState {
    Idle,
    Running,
    Paused,
    Degraded,
    Error,
}

impl TrayState {
    fn icon_path(&self) -> &'static str {
        match self {
            TrayState::Idle     => "icons/tray_idle.png",
            TrayState::Running   => "icons/tray_running.png",
            TrayState::Paused    => "icons/tray_paused.png",
            TrayState::Degraded  => "icons/tray_degraded.png",
            TrayState::Error     => "icons/tray_error.png",
        }
    }
}

/// TrayManager handles tray icon, menu, and event routing
pub struct TrayManager {
    pub state: StdMutex<TrayState>,
}

impl TrayManager {
    pub fn new() -> Self {
        Self {
            state: StdMutex::new(TrayState::Idle),
        }
    }

    pub fn get_state(&self) -> TrayState {
        *self.state.lock().unwrap()
    }

    pub fn set_state(&self, new_state: TrayState, app: &tauri::AppHandle) {
        if let Some(tray) = app.tray_by_id("main-tray") {
            let icon_path = new_state.icon_path();
            if let Ok(icon) = Image::from_path(icon_path) {
                let _ = tray.set_icon(Some(icon));
            }
            *self.state.lock().unwrap() = new_state;
        }
    }
}

// Tray menu builder
fn build_tray_menu(app: &tauri::AppHandle) -> tauri::Result<Menu<tauri::Wry>> {
    use tauri::menu::{MenuBuilder, MenuItemBuilder, PredefinedMenuItemBuilder};

```rust
// Tray setup
let app_handle = app.handle().clone();
let tray_app = app_handle.clone();
let tray_manager_setup = tray_manager.clone();

// Build tray menu
let menu = build_tray_menu(&tray_app)?;

// Create tray icon
let icon = Image::from_path("icons/tray_idle.png")
    .map_err(|e| format!("Failed to load tray icon: {}", e))?;

let _tray = TrayIconBuilder::with_id("main-tray")
    .icon(icon)
    .menu(&menu)
    .tooltip("Zedsu — IDLE")
    .on_menu_event(move |app, event| {
        match event.id().as_ref() {
            "tray_start" => {
                if let Some(bm) = app.try_state::<Arc<Mutex<BackendManager>>>() {
                    if let Ok(mgr) = bm.lock() {
                        let _ = mgr.send_command("start", None);
                    }
                }
            }
            "tray_stop" => {
                if let Some(bm) = app.try_state::<Arc<Mutex<BackendManager>>>() {
                    if let Ok(mgr) = bm.lock() {
                        let _ = mgr.send_command("stop", None);
                    }
                }
            }
            "tray_pause" => {
                if let Some(bm) = app.try_state::<Arc<Mutex<BackendManager>>>() {
                    if let Ok(mgr) = bm.lock() {
                        let _ = mgr.send_command("pause", None);
                    }
                }
            }
            "tray_resume" => {
                if let Some(bm) = app.try_state::<Arc<Mutex<BackendManager>>>() {
                    if let Ok(mgr) = bm.lock() {
                        let _ = mgr.send_command("resume", None);
                    }
                }
            }
            "tray_estop" => {
                if let Some(bm) = app.try_state::<Arc<Mutex<BackendManager>>>() {
                    if let Ok(mgr) = bm.lock() {
                        let _ = mgr.send_command("emergency_stop", None);
                    }
                }
            }
            "tray_toggle_hud" => {
                if let Some(hud) = app.get_webview_window("hud") {
                    if hud.is_visible().unwrap_or(false) {
                        let _ = hud.hide();
                    } else {
                        let _ = hud.show();
                        let _ = hud.set_focus();
                    }
                }
            }
            "tray_open_shell" => {
                if let Some(main) = app.get_webview_window("main") {
                    let _ = main.show();
                    let _ = main.set_focus();
                }
            }
            "tray_open_logs" => {
                let logs_path = std::env::current_exe()
                    .ok()
                    .and_then(|p| p.parent().map(|p| p.join("logs")))
                    .map(|p| p.to_string_lossy().to_string())
                    .unwrap_or_default();
                if !logs_path.is_empty() {
                    let _ = std::process::Command::new("explorer")
                        .arg(&logs_path)
                        .spawn();
                }
            }
            "tray_restart" => {
                if let Some(bm) = app.try_state::<Arc<Mutex<BackendManager>>>() {
                    if let Ok(mut mgr) = bm.lock() {
                        let _ = mgr.respawn();
                    }
                }
            }
            "tray_quit" => {
                // Graceful shutdown: stop backend first
                if let Some(bm) = app.try_state::<Arc<Mutex<BackendManager>>>() {
                    if let Ok(mgr) = bm.lock() {
                        let _ = mgr.send_command("stop", None);
                        std::thread::sleep(std::time::Duration::from_millis(200));
                        mgr.stop();
                    }
                }
                app.exit(0);
            }
            _ => {}
        }
    })
    .on_tray_icon_event(move |tray, event| {
        match event {
            TrayIconEvent::Click { button: MouseButton::Left, button_state: MouseButtonState::Up, .. } => {
                // Left click: toggle shell visibility
                if let Some(main) = tray.app_handle().get_webview_window("main") {
                    if main.is_visible().unwrap_or(false) {
                        let _ = main.hide();
                    } else {
                        let _ = main.show();
                        let _ = main.set_focus();
                    }
                }
            }
            TrayIconEvent::DoubleClick { .. } => {
                // Double click: open Overview (show main window)
                if let Some(main) = tray.app_handle().get_webview_window("main") {
                    let _ = main.show();
                    let _ = main.set_focus();
                }
            }
            _ => {}
        }
    })
    .build(app)?;
```

/// Position HUD window at top-right of primary monitor with margin.
/// Call this once at startup and when corner setting changes.
pub fn position_hud_top_right(app: &tauri::AppHandle) {
    if let Some(hud) = app.get_webview_window("hud") {
        // Get primary monitor dimensions via DPI API
        // On Windows: use EnumDisplayMonitors + GetMonitorInfoW
        // Fallback: use screen size from window
        let margin: i32 = 16;

        if let Ok(monitor_info) = get_primary_monitor_info() {
            let x = monitor_info.left + monitor_info.width - 360 - margin;
            let y = monitor_info.top + margin;
            let _ = hud.set_position(tauri::PhysicalPosition::new(x, y));
        } else {
            // Fallback: position at right side of virtual screen
            let _ = hud.set_position(tauri::PhysicalPosition::new(
                (1920 - 360 - margin) as i32,
                margin as i32,
            ));
        }
    }
}
