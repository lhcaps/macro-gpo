**CRITICAL backend contract (verified from `src/zedsu_backend.py` do_POST):**
- Command body: `{ action: "update_config", payload: updates }` — NOT `{ command, config }`
- `/state` does NOT include top-level `regions`, `positions`, `combat_ai` — fetch these via `/command`
- `combat_ai` config is FLAT keys inside `config.combat_ai`: `telemetry_enabled`, `target_lost_grace_sec`, `crowd_risk_threshold`, etc. — NOT nested objects
- `test_discord_webhook` takes NO arguments — backend reads saved URL internally
- `set_region` payload: `{ name, area: [x1,y1,x2,y2], kind, threshold, enabled, label }`
- `set_position` payload: `{ name, x, y, label, enabled, captured_at, window_title }`
- `test_webhook` response: `{ status: 'ok', has_webhook, sent }`

This plan builds 7 page modules. Each page is loaded dynamically when the user navigates to the corresponding sidebar item.

Architecture:
- `src/scripts/shared/config-api.js` — shared config fetch/save helpers
- `src/scripts/pages/{page}.js` — one JS file per page
- `src/styles/pages/{page}.css` — one CSS file per page (optional, can reuse app.css)
- Page modules receive state from `zedsu:state-update` events or call `ShellApi.getState()`

```javascript
// config-api.js — Shared backend API helpers for all settings pages
// CRITICAL: All commands use { action, payload } format per backend contract

const API_BASE = 'http://localhost:9761';

// ============================================================
// Base command helper
// Backend expects: { action: string, payload: any }
// Backend responds: { status: 'ok'|'error', ... }
// ============================================================

async function postCommand(action, payload = null) {
  const resp = await fetch(`${API_BASE}/command`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ action, payload }),
  });
  if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
  const data = await resp.json();
  if (data.status === 'error') throw new Error(data.message || 'Command failed');
  return data;
}

// ============================================================
// Config
// ============================================================

/**
 * Get current sanitized config from backend.
 * Returns: { config: {...} } with webhook URL stripped.
 */
export async function getConfig() {
  const resp = await fetch(`${API_BASE}/state`);
  if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
  const data = await resp.json();
  return data.config || {};
}

/**
 * Update config via deep-merge.
 * Payload is the partial config to merge.
 */
export async function updateConfig(updates) {
  return postCommand('update_config', updates);
}

// ============================================================
// Regions — NOT in /state; use /command
// ============================================================

export async function getRegions() {
  const data = await postCommand('get_regions');
  return data.regions || [];
}

export async function setRegion(name, region) {
  // Backend expects payload: { name, area: [x1,y1,x2,y2], kind, threshold, enabled, label }
  return postCommand('set_region', { name, ...region });
}

export async function resolveRegion(name) {
  return postCommand('resolve_region', { name });
}

export async function deleteRegion(name) {
  return postCommand('delete_region', { name });
}

// ============================================================
// Positions — NOT in /state; use /command
// ============================================================

export async function getPositions() {
  const data = await postCommand('get_positions');
  return data.positions || [];
}

export async function setPosition(name, position) {
  // Backend expects payload: { name, x, y, label, enabled, captured_at, window_title }
  return postCommand('set_position', { name, ...position });
}

export async function resolvePosition(name) {
  return postCommand('resolve_position', { name });
}

export async function deletePosition(name) {
  return postCommand('delete_position', { name });
}

// ============================================================
// Discord — webhook URL managed by backend, NEVER send URL from frontend
// ============================================================

export async function testDiscordWebhook() {
  // Takes NO arguments — backend reads saved URL internally
  return postCommand('test_discord_webhook');
}

export async function updateDiscordConfig(updates) {
  return postCommand('update_config', { discord_events: updates });
}

// ============================================================
// YOLO
// ============================================================

export async function sendYoloCommand(cmd, params = {}) {
  return postCommand(cmd, params);
}

// ============================================================
// State normalization — /state fields
// NOTE: /state does NOT have top-level regions/positions/combat_ai
// Regions and positions must be fetched via /command
// ============================================================

/**
 * Normalize backend /state response to UI contract.
 * /state gives: config, combat (stats), hud (runtime), yolo_model, logs
 * NOT: regions, positions, combat_ai (those come from config or commands)
 */
export function normalizeState(raw) {
  const config = raw?.config || {};
  const combat = raw?.combat || {};
  const hud = raw?.hud || {};
  const yolo = raw?.yolo_model || {};

  return {
    running: !!raw?.running,
    status: raw?.status || 'Unknown',
    hud,
    yolo,
    logs: raw?.logs || [],
    config,

    // Combat stats from /state.combat (not /state.combat_ai)
    combat: {
      kills: combat.kills || 0,
      match_num: combat.match_num || null,
      elapsed: hud.elapsed_sec || 0,
      last_event: combat.last_event || null,
      death_reason: combat.death_reason || null,
      combat_state: combat.combat_state || null,
      intent: combat.intent || null,
      crowd_risk: combat.crowd_risk || null,
    },

    // combat_ai config from config.combat_ai (FLAT keys)
    combatAI: {
      config: config.combat_ai || {},
    },

    // Discord from config.discord_events
    discord: config.discord_events || {},
  };
}

// ============================================================
// Utility
// ============================================================

export function escapeHtml(str) {
  if (str === null || str === undefined) return '';
  const div = document.createElement('div');
  div.textContent = String(str);
  return div.innerHTML;
}
```

// detection.js — Combat Detection settings page

const REGIONS = ['enemy_hp_bar', 'damage_numbers', 'player_hp_bar', 'incombat_timer', 'kill_icon'];

export async function load(container) {
  container.innerHTML = '<div class="page-loading"><div class="loading-spinner"></div></div>';

          window.ShellApi?.Toast.success(`Test result: ${result.detected ? 'Combat DETECTED' : 'No combat detected'}`);
        } else {
          window.ShellApi?.Toast.error('Test failed: ' + (result.error || 'unknown'));
        }
      },

      async pickRegion(name) {
        window.ShellApi?.Toast.info(`Starting region picker for: ${name}`);
        const result = await api.sendYoloCommand('select_region', { name });
        if (result.success) {
          window.ShellApi?.Toast.success(`${name} region set`);
          load(container); // Refresh
        } else {
          window.ShellApi?.Toast.error('Region selection cancelled or failed');
        }
      },

      async testRegion(name) {
        const result = await api.resolveRegion(name);
        if (result.success) {
          window.ShellApi?.Toast.success(`${name}: ${JSON.stringify(result.region || result.result)}`);
        } else {
          window.ShellApi?.Toast.error(`Test failed: ${result.error || 'unknown'}`);
        }
      },

      async resetRegion(name) {
        if (!confirm(`Reset region "${name}"? This will remove it.`)) return;
        const result = await api.deleteRegion(name);
        if (result.success) {
          window.ShellApi?.Toast.success(`${name} reset`);
          load(container);
        } else {
          window.ShellApi?.Toast.error('Failed to reset region');
        }
      },
    };

  } catch (err) {
    container.innerHTML = `<div class="page-error"><p>Failed to load Detection page: ${escapeHtml(err.message)}</p></div>`;
  }
}

function escapeHtml(str) {
  const div = document.createElement('div');
  div.textContent = str;
  return div.innerHTML;
}

// logs.js — Backend logs viewer page

const API_BASE = 'http://localhost:9761';
let container = null;
let allLogs = [];
let pollTimer = null;
let currentFilter = 'all'; // all, info, warn, error

export async function load(target) {
  container = target;
  container.innerHTML = `
    <div class="logs-page">
      <div class="logs-toolbar">
        <div class="logs-filters">
          <button class="btn btn-xs filter-btn active" data-filter="all">All</button>
          <button class="btn btn-xs filter-btn" data-filter="info">Info</button>
          <button class="btn btn-xs filter-btn" data-filter="warn">Warning</button>
          <button class="btn btn-xs filter-btn" data-filter="error">Error</button>
        </div>
        <div class="logs-actions">
          <button class="btn btn-xs" onclick="window.__logsPage?.copyLogs()">Copy</button>
          <button class="btn btn-xs" onclick="window.__logsPage?.openLogsFolder()">Open Folder</button>
          <button class="btn btn-xs" onclick="window.__logsPage?.clearLogs()">Clear View</button>
        </div>
      </div>
      <div class="logs-viewer" id="logs-viewer">
        <div class="page-loading"><div class="loading-spinner"></div></div>
      </div>
    </div>
  `;

  // Filter buttons
  container.querySelectorAll('.filter-btn').forEach(btn => {
    btn.addEventListener('click', () => {
      currentFilter = btn.getAttribute('data-filter');
      container.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      renderLogs();
    });
  });

  // Initial load
  await fetchLogs();
  pollTimer = setInterval(fetchLogs, 2000);

  window.__logsPage = {
    copyLogs() {
      const filtered = filterLogs();
      const text = filtered.map(l => `[${l.timestamp || ''}] [${l.level || 'INFO'}] ${l.message || l}`).join('\n');
      navigator.clipboard.writeText(text).then(() => {
        window.ShellApi?.Toast.success('Logs copied to clipboard');
      }).catch(() => {
        window.ShellApi?.Toast.error('Failed to copy logs');
      });
    },

    openLogsFolder() {
      try {
        window.open('./logs/', '_blank');
      } catch {
        window.ShellApi?.Toast.info('Logs folder: ./logs/');
      }
    },

    clearLogs() {
      allLogs = [];
      renderLogs();
    },
  };

  return () => {
    if (pollTimer) clearInterval(pollTimer);
  };
}

async function fetchLogs() {
  try {
    const resp = await fetch(`${API_BASE}/state`);
    if (!resp.ok) return;
    const data = await resp.json();
    const logs = data.logs || [];
    if (logs.length > 0) {
      allLogs = logs;
      renderLogs();
    }
  } catch {
    // Backend offline
  }
}

function filterLogs() {
  if (currentFilter === 'all') return allLogs;
  return allLogs.filter(l => {
    const level = (l.level || 'INFO').toLowerCase();
    if (currentFilter === 'warn') return level === 'warn' || level === 'warning';
    if (currentFilter === 'error') return level === 'error' || level === 'critical';
    return level === currentFilter;
  });
}

function renderLogs() {
  const viewer = document.getElementById('logs-viewer');
  if (!viewer) return;

  const filtered = filterLogs();
  if (filtered.length === 0) {
    viewer.innerHTML = '<p class="logs-empty">No logs to display.</p>';
    return;
  }

  viewer.innerHTML = filtered.map(l => {
    const level = (l.level || 'INFO').toLowerCase();
    const levelClass = level === 'error' || level === 'critical' ? 'log-error' :
                       level === 'warn' || level === 'warning' ? 'log-warn' : 'log-info';
    const time = l.timestamp ? new Date(l.timestamp).toLocaleTimeString() : '';
    return `<div class="log-line ${levelClass}">
      <span class="log-time">${time}</span>
      <span class="log-level">${(l.level || 'INFO').toUpperCase().padEnd(7)}</span>
      <span class="log-message">${escapeHtml(l.message || JSON.stringify(l))}</span>
    </div>`;
  }).join('');

  // Auto-scroll to bottom
  viewer.scrollTop = viewer.scrollHeight;
}

function escapeHtml(str) {
  const div = document.createElement('div');
  div.textContent = str;
  return div.innerHTML;
}
