## Task 5: Verify Settings update_config persists

<action>

Test that the Settings pages correctly call `update_config` and it persists:

```javascript
// Test flow (manual or write a test script):
// 1. Open Discord page
// 2. Enter a webhook URL
// 3. Click Save
// 4. Verify backend received the update
// 5. Verify the UI shows "CONFIGURED" not the actual URL
// 6. Refresh page
// 7. Verify webhook is still configured (not the URL, just the "configured" state)
```

Create a test script `src/ZedsuFrontend/test-settings.js`:

```javascript
// test-settings.js — Smoke test for settings persistence
// Run with: node test-settings.js (requires node-fetch)

const fetch = require('node-fetch');

const BASE = 'http://localhost:9761';

async function test() {
  // Test 1: update_config persists
  console.log('Test 1: update_config...');
  const resp = await fetch(`${BASE}/command`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      command: 'update_config',
      config: { runtime: { scan_interval: 150 } },
    }),
  });
  const data = await resp.json();
  console.log('  update_config result:', data.success ? 'OK' : 'FAIL');

  // Test 2: config reflects change
  console.log('Test 2: config reflects change...');
  const state = await fetch(`${BASE}/state`).then(r => r.json());
  const scanInterval = state.config?.runtime?.scan_interval;
  console.log('  scan_interval:', scanInterval, scanInterval === 150 ? 'OK' : 'FAIL');

  // Test 3: webhook sanitization
  console.log('Test 3: webhook sanitization...');
  const hasWebhookUrl = JSON.stringify(state).includes('discord.com/api/webhooks');
  console.log('  webhook URL in state:', hasWebhookUrl ? 'LEAK!' : 'OK (sanitized)');

  // Test 4: discord config
  console.log('Test 4: discord config...');
  const discordResp = await fetch(`${BASE}/command`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      command: 'update_config',
      config: { discord_events: { webhook_url: 'https://discord.com/api/webhooks/test/test' } },
    }),
  });
  const discordResult = await discordResp.json();
  console.log('  discord update:', discordResult.success ? 'OK' : 'FAIL');

  // Verify webhook still sanitized
  const state2 = await fetch(`${BASE}/state`).then(r => r.json());
  const webhookInState2 = JSON.stringify(state2).includes('discord.com/api/webhooks');
  console.log('  webhook sanitized after save:', webhookInState2 ? 'LEAK!' : 'OK');

  console.log('\nAll tests complete.');
}

test().catch(console.error);
```

Run the test:
```bash

```

</action>
