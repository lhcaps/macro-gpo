The new block must implement the **blocking overlay thread pattern**:

```python
elif action == "select_region":
    # Blocking overlay pattern: dedicated non-daemon thread, HTTP handler blocks on result.
    # This is different from YOLO capture (daemon, fire-and-forget).
    from src.utils.windows import get_window_rect
    from src.utils.config import save_config, load_config
    from src.services.region_service import set_region
    from src.overlays.region_selector import RegionSelectorOverlay

    payload_data = data.get("payload", {})
    region_name = payload_data.get("name", "combat_scan")

    window_title = _app_config.get("game_window_title", "")
    if not window_title:
        self._send_json({"status": "error", "message": "No game_window_title configured"}, 400)
        return

    rect = get_window_rect(window_title)
    if rect is None:
        self._send_json({"status": "error", "message": f"Window not found: {window_title}"}, 404)
        return

    _backend_log.info(f"[SELECT_REGION] Starting overlay for region=%s window=%s", region_name, window_title)

    # Create overlay and dedicated thread (non-daemon so HTTP handler can join it)
    overlay = RegionSelectorOverlay(window_title, region_name)
    overlay_done = threading.Event()

    def _run_overlay():
        try:
            overlay.run()
            result = overlay.get_result(timeout=300.0)  # 5 min max
        except Exception as e:
            _backend_log.error(f"[SELECT_REGION] Overlay error: {e}")
            result = {"action": "cancel"}
        finally:
            overlay_done.set()
        # Process result AFTER overlay closes
        if result is None:
            _backend_log.warning("[SELECT_REGION] Overlay timed out")
            return
        if result.get("action") == "cancel":
            _backend_log.info("[SELECT_REGION] Region selection cancelled")
            return
        area = result.get("area", [])
        if not area:
            _backend_log.warning("[SELECT_REGION] No area in result")
            return
        # Normalize and save
        success, err = set_region(_app_config, region_name, area, kind="generic")
        if not success:
            _backend_log.error(f"[SELECT_REGION] set_region failed: {err}")
            return
        save_config(_app_config)
        _app_config = load_config()
        _backend_log.info(f"[SELECT_REGION] Region saved: {region_name} area={area}")

    # NON-daemon thread -- HTTP handler will join() this thread
    overlay_thread = threading.Thread(target=_run_overlay, name="RegionSelector")
    overlay_thread.start()

    # Block here until overlay thread completes
    overlay_thread.join(timeout=310.0)

    # Now send the response (after overlay has closed)
    # The config is already persisted if user pressed Enter
    self._send_json({"status": "ok", "region": region_name})
```

Key implementation notes:
- **Non-daemon thread**: `threading.Thread(..., name="RegionSelector")` -- no `daemon=True`. This is critical. If the thread is daemon, the HTTP server will kill it when other requests complete. The HTTP handler MUST `join()` the thread.
- **Imports inside handler**: All imports (`get_window_rect`, `save_config`, `load_config`, `set_region`, `RegionSelectorOverlay`) are handler-level, not module-level.
- **Blocking pattern**: `overlay_thread.join(timeout=310.0)` blocks the HTTP handler thread. This means one HTTP request is blocked while the overlay is open. The HTTP server uses ThreadingHTTPServer so other requests can still be handled by other threads.
- **Fallback join**: Even if the 5-min timeout expires inside `_run_overlay`, the HTTP handler's `join(timeout=310.0)` will return, and the handler sends the response.
