    def show_notification(self, title, message, timeout=5):
        """Show Windows balloon notification via Shell_NotifyIcon."""
        try:
            import ctypes
            from ctypes import wintypes
            
            # Load shell32.dll
            shell32 = ctypes.windll.shell32
            
            # NOTIFYICONDATA structure (Windows 7+)
            class NOTIFYICONDATA(ctypes.Structure):
                _fields_ = [
                    ("cbSize", wintypes.DWORD),
                    ("hWnd", wintypes.HWND),
                    ("uID", wintypes.UINT),
                    ("uFlags", wintypes.UINT),
                    ("uCallbackMessage", wintypes.UINT),
                    ("hIcon", wintypes.HICON),
                    ("szTip", wintypes.WCHAR * 128),
                    ("dwState", wintypes.DWORD),
                    ("dwStateMask", wintypes.DWORD),
                    ("szInfo", wintypes.WCHAR * 256),
                    ("uTimeout", wintypes.UINT),
                    ("szInfoTitle", wintypes.WCHAR * 64),
                    ("dwInfoFlags", wintypes.DWORD),
                ]
            
            NIF_INFO = 0x10
            NIIF_INFO = 0x1
            
            # Get window handle
            hwnd = self.app.winfo_id() if hasattr(self.app, 'winfo_id') else None
            if not hwnd:
                return
            
            nid = NOTIFYICONDATA()
            nid.cbSize = ctypes.sizeof(NOTIFYICONDATA)
            nid.hWnd = hwnd
            nid.uID = 1
            nid.uFlags = NIF_INFO
            nid.szInfoTitle = title
            nid.szInfo = message
            nid.uTimeout = timeout * 1000
            nid.dwInfoFlags = NIIF_INFO
            
            # Get window title for tooltip (used as app ID)
            # This is a simplified version
            shell32.Shell_NotifyIconW(0x1, ctypes.byref(nid))  # NIM_MODIFY
            
        except Exception as e:
            # Fallback: just log
            print(f"[TrayManager] Notification failed: {e}")
