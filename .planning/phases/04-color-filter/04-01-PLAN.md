    Add a new function `_hsv_color_check()` to vision.py:
    
    def _hsv_color_check(img_name, haystack_bgr, config):
        """Fast HSV color range check. Returns True if color found, False otherwise.
        Does NOT return bounding box — only presence detection."""
        hsv_config = config.get("hsv_ranges", {}).get(img_name, {})
        if not hsv_config.get("enabled", False):
            return False
        
        try:
            import cv2
            import numpy as np
        except ImportError:
            return False
        
        lower = np.array(hsv_config["lower"], dtype=np.uint8)
        upper = np.array(hsv_config["upper"], dtype=np.uint8)
        
        # Convert BGR haystack to HSV
        if len(haystack_bgr.shape) == 3:
            hsv = cv2.cvtColor(haystack_bgr, cv2.COLOR_BGR2HSV)
        else:
            return False
        
        # Create mask
        mask = cv2.inRange(hsv, lower, upper)
        
        # For red hue (wrap-around): if lower[0] > upper[0], do two checks
        if lower[0] > upper[0]:
            lower2 = np.array([0, lower[1], lower[2]], dtype=np.uint8)
            upper2 = np.array([upper[0], upper[1], upper[2]], dtype=np.uint8)
            mask2 = cv2.inRange(hsv, lower2, upper2)
            mask = cv2.bitwise_or(mask, mask2)
        
        # Count non-zero pixels in mask
        pixel_count = cv2.countNonZero(mask)
        
        # Threshold: at least 0.05% of total pixels must match
        min_pixels = max(10, int(haystack_bgr.shape[0] * haystack_bgr.shape[1] * 0.0005))
        
        return pixel_count >= min_pixels
    
    NOTE: This function returns True/False only — no bounding box.
    The template match that follows (if color check passes) will provide the box.
  </action>
