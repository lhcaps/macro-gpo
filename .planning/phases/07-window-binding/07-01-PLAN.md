            self._last_window_rect = current_rect
            return
        
        # Compare dimensions
        old_l, old_t, old_r, old_b = self._last_window_rect
        new_l, new_t, new_r, new_b = current_rect
        
        old_w, old_h = old_r - old_l, old_b - old_t
        new_w, new_h = new_r - new_l, new_b - new_t

            return
