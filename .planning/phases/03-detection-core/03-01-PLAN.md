    def _mss_capture_haystack(normalized_region):
        """Fast screen capture using MSS. Returns (image_rgb, offset) tuple."""
        if not _MSS_AVAILABLE:
            return None, (0, 0)
        try:
            with mss.mss() as sct:
                monitor = {
                    "left": int(normalized_region[0]),
                    "top": int(normalized_region[1]),
                    "width": int(normalized_region[2]),
                    "height": int(normalized_region[3]),
                }
                screenshot = sct.grab(monitor)
                # BGRA -> RGB (PIL-compatible format)
                img = np.array(screenshot)
                if img.shape[2] == 4:
                    img = img[:, :, :3][:, :, ::-1]  # BGRA -> RGB
                return img, (monitor["left"], monitor["top"])
        except Exception:
            return None, (0, 0)

    def _cv2_locate_image(img_name, config, confidence=None, region=None, search_context=None):
        """Fast image location using cv2.matchTemplate + MSS capture."""
        path = resolve_path(config.get("images", {}).get(img_name))
        if not path or not os.path.exists(path):
            return None
        
        conf = float(confidence if confidence is not None else config.get("confidence", 0.8))
        use_mss = config.get("detection_backend", "auto") != "pyautogui"
        
        # Capture: use MSS if available and auto/opencv mode
        if use_mss and _MSS_AVAILABLE and search_context and search_context.get("region"):
            haystack_rgb = search_context.get("image")
            offset = search_context.get("offset", (0, 0))
        elif use_mss and _MSS_AVAILABLE:
            region_rect = get_window_rect(str(config.get("game_window_title", "")))
            if region_rect:
                normalized = _normalize_region(region_rect)
                haystack_rgb, offset = _mss_capture_haystack(normalized)
            else:
                haystack_rgb, offset = _capture_haystack(None)  # fallback to PIL
        else:
            haystack_rgb, offset = _capture_haystack(region)
        
        if haystack_rgb is None:
            return None
        
        # Convert to grayscale
        if len(haystack_rgb.shape) == 3:
            haystack_gray = cv2.cvtColor(haystack_rgb, cv2.COLOR_RGB2GRAY)
        else:
            haystack_gray = haystack_rgb
        
        # Build scale candidates (same logic as existing)
        current_size = _region_size(region) or [haystack_rgb.shape[1], haystack_rgb.shape[0]]
        scales = _build_scale_candidates(img_name, config, current_size)
        
        # Try each scale and grayscale option
        for scale in scales[:6]:
            needle = _scaled_template(path, scale)
            needle_gray = cv2.cvtColor(needle, cv2.COLOR_RGB2GRAY) if len(needle.shape) == 3 else needle
            
            for attempt_conf in [conf, max(0.58, conf - 0.04), max(0.5, conf - 0.12)]:
                try:
                    result = cv2.matchTemplate(
                        haystack_gray, needle_gray, cv2.TM_CCOEFF_NORMED
                    )
                    _, max_val, _, max_loc = cv2.minMaxLoc(result)
                    
                    if max_val >= attempt_conf:
                        _SCALE_HINT_CACHE[img_name] = scale
                        w, h = needle.shape[1], needle.shape[0]
                        # Convert to pyautogui-compatible box format
                        class FakeBox:
                            pass
                        box = FakeBox()
                        box.left = max_loc[0] + offset[0]
                        box.top = max_loc[1] + offset[1]
                        box.width = w
                        box.height = h
                        _register_last_match(img_name, box, {"region": region})
                        return box
                except Exception:
                    continue
        
        return None
